-- 1. SCHOOLS TABLE (The Master List) CREATE TABLE Schools ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), sch_id TEXT UNIQUE NOT NULL, -- e.g., PM001 school_name TEXT NOT NULL, admin_key TEXT NOT NULL, -- Admin Password teacher_key TEXT NOT NULL, -- Teacher Password finance_key TEXT NOT NULL, -- Finance Password expiry_date DATE NOT NULL, -- Auto-shutdown date is_active BOOLEAN DEFAULT true, created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW() ); -- 2. STUDENTS TABLE (The Registry) CREATE TABLE Students ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), sch_id TEXT REFERENCES Schools(sch_id), name TEXT NOT NULL, photo_url TEXT, dob DATE, gender TEXT, class TEXT NOT NULL, -- Baby Class to S.6 lin TEXT UNIQUE, -- Format: PM-J-4821 location TEXT, health_status TEXT, contact_number TEXT, total_fees_required DECIMAL DEFAULT 0, fees_paid DECIMAL DEFAULT 0, created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW() ); -- 3. ATTENDANCE TABLE CREATE TABLE Attendance ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), student_id UUID REFERENCES Students(id), sch_id TEXT REFERENCES Schools(sch_id), status TEXT CHECK (status IN ('Present', 'Absent')), date DATE DEFAULT CURRENT_DATE ); -- 4. MARKS TABLE (For Report Cards) CREATE TABLE Marks ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), student_id UUID REFERENCES Students(id), sch_id TEXT REFERENCES Schools(sch_id), subject TEXT, score INTEGER, term TEXT, is_approved BOOLEAN DEFAULT false -- Admin must toggle this );
